const apiKey = "YOUR_API_KEY"; const cities = ["Paris", "Tokyo", "New York"]; // Travel score (based on weather) function calculateTravelScore(temp) { if (temp >= 20 && temp <= 30) return 90; if (temp >= 10 && temp < 20) return 75; if (temp > 30) return 60; return 50; } // Food score (simulated intelligence) function getFoodScore(city) { const foodScores = { "Paris": 95, "Tokyo": 98, "New York": 90 }; return foodScores[city] || 80; } async function getCityData(city) { const response = await fetch( `https://api.openweathermap.org/data/2.5/weather?q=${city}&appid=${apiKey}&units=metric` ); const data = await response.json(); const temp = data.main.temp; const weather = data.weather[0].description; const travelScore = calculateTravelScore(temp); const foodScore = getFoodScore(city); // Combine both const overallIndex = Math.round((travelScore * 0.6) + (foodScore * 0.4)); return ` ${city}
🌤 Weather: ${temp}°C, ${weather}
🍽 Food Index: ${foodScore}
✈ Travel Index: ${travelScore}
📊 Overall Insight Index: ${overallIndex} `; } async function loadData() { let results = []; for (let city of cities) { const result = await getCityData(city); results.push(result); } document.getElementById("city").innerHTML = results.join("

"); } loadData();
