const apiKey = "YOUR_API_KEY"; const cities = ["Paris", "Tokyo", "New York"]; // Travel score function calculateTravelScore(temp) { if (temp >= 20 && temp <= 30) return 90; if (temp >= 10 && temp < 20) return 75; if (temp > 30) return 60; return 50; } // Food score function getFoodScore(city) { const scores = { "Paris": 95, "Tokyo": 98, "New York": 90 }; return scores[city] || 80; } // AI Insight function function generateInsight(city, temp, weather, travelScore, foodScore) { if (travelScore > 80 && foodScore > 90) { return city + " shows strong overall appeal with favorable weather and excellent food culture."; } if (travelScore > 70) { return city + " presents moderate travel conditions with reasonably supportive weather patterns."; } return city + " currently reflects limited travel comfort due to less favorable environmental conditions."; } async function getCityData(city) { try { const response = await fetch( "https://api.openweathermap.org/data/2.5/weather?q=" + city + "&appid=" + apiKey + "&units=metric" ); const data = await response.json(); if (data.cod !== 200) { return city + ": Error loading data"; } const temp = data.main.temp; const weather = data.weather[0].description; const travelScore = calculateTravelScore(temp); const foodScore = getFoodScore(city); const overall = Math.round((travelScore * 0.6) + (foodScore * 0.4)); const insight = generateInsight(city, temp, weather, travelScore, foodScore); return ` ${city}
🌤 ${temp}°C, ${weather}
🍽 Food Index: ${foodScore}
✈ Travel Index: ${travelScore}
📊 Composite Travel Indicator: ${overall}
🤖 Insight: ${insight} `; } catch (error) { return city + ": Error fetching data"; } } async function loadData() { let results = []; for (let city of cities) { const result = await getCityData(city); results.push(result); } document.getElementById("city").innerHTML = results.join("

"); } loadData();
